\item example: defn of convergence of a sequence

\begin{minted}[]{lean}
-- definition of "u tends to ℓ" 
def seq_limit (u : ℕ → ℝ) (ℓ : ℝ) :=
  ∀ ε > 0, ∃ N, ∀ n ≥ N, |u n - ℓ| ≤ ε
\end{minted}
